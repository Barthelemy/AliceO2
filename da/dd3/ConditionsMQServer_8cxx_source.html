59  while (CheckCurrentState(RUNNING)) {
60  std::unique_ptr<FairMQMessage> req(fTransportFactory->CreateMessage());
61 
62  if (fChannels.at("data").at(0).Receive(req) > 0) {
63  std::string reqStr(static_cast<char*>(req->GetData()), req->GetSize());
64  LOG(INFO) << "Received parameter request from client: \"" << reqStr << "\"";
65 
66  size_t pos = reqStr.rfind(",");
67  string newParameterName = reqStr.substr(0, pos);
68  int runId = std::stoi(reqStr.substr(pos + 1));
69  LOG(INFO) << "Parameter name: " << newParameterName;
70  LOG(INFO) << "Run ID: " << runId;
71 
72  LOG(INFO) << "Retrieving parameter...";
73  fCdbManager->setRun(runId);
74  aCondition = fCdbManager->getObject(IdPath(newParameterName), runId);
75 
76  if (aCondition) {
77  LOG(INFO) << "Sending following parameter to the client:";
78  aCondition->printConditionMetaData();
79  TMessage* tmsg = new TMessage(kMESS_OBJECT);
80  tmsg->WriteObject(aCondition);
81 
82  std::unique_ptr<FairMQMessage> reply(
83  fTransportFactory->CreateMessage(tmsg->Buffer(), tmsg->BufferSize(), free_tmessage, tmsg));
84 
85  fChannels.at("data").at(0).Send(reply);
86  }
87  else {
88  LOG(ERROR) << "Could not get a condition for \"" << parameterName << "\" and run " << runId << "!";
89  }
90  }
91  }
92 }
