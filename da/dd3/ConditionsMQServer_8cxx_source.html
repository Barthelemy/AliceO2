99 void ConditionsMQServer::Run()
100 {
101  std::string parameterName = "";
102  FairParGenericSet* par = nullptr;
103  Condition* aCondition = nullptr;
104 
105  while (CheckCurrentState(RUNNING)) {
106  std::unique_ptr<FairMQMessage> req(fTransportFactory->CreateMessage());
107 
108  if (fChannels.at("data").at(0).Receive(req) > 0) {
109  std::string reqStr(static_cast<char*>(req->GetData()), req->GetSize());
110  LOG(INFO) << "Received parameter request from client: \"" << reqStr << "\"";
111 
112  size_t pos = reqStr.rfind(",");
113  string newParameterName = reqStr.substr(0, pos);
114  int runId = std::stoi(reqStr.substr(pos + 1));
115  LOG(INFO) << "Parameter name: " << newParameterName;
116  LOG(INFO) << "Run ID: " << runId;
117 
118  LOG(INFO) << "Retrieving parameter...";
119  // Check if the parameter name has changed to avoid getting same container repeatedly
120  if (fFirstInputType != "OCDB") {
121  if (newParameterName != parameterName) {
122  parameterName = newParameterName;
123  par = static_cast<FairParGenericSet*>(fRtdb->getContainer(parameterName.c_str()));
124  }
125  fRtdb->initContainers(runId);
126  }
127  else {
128  fCdbManager->setRun(runId);
129  aCondition = fCdbManager->getObject(IdPath(newParameterName), runId);
130  }
131 
132  LOG(INFO) << "Sending following parameter to the client:";
133  if (par) {
134  par->print();
135 
136  TMessage* tmsg = new TMessage(kMESS_OBJECT);
137  tmsg->WriteObject(par);
138 
139  std::unique_ptr<FairMQMessage> reply(
140  fTransportFactory->CreateMessage(tmsg->Buffer(), tmsg->BufferSize(), free_tmessage, tmsg));
141 
142  fChannels.at("data").at(0).Send(reply);
143  }
144  else if (aCondition) {
145  aCondition->printConditionMetaData();
146  TMessage* tmsg = new TMessage(kMESS_OBJECT);
147  tmsg->WriteObject(aCondition);
148 
149  std::unique_ptr<FairMQMessage> reply(
150  fTransportFactory->CreateMessage(tmsg->Buffer(), tmsg->BufferSize(), free_tmessage, tmsg));
151 
152  fChannels.at("data").at(0).Send(reply);
153  }
154  else {
155  LOG(ERROR) << "Parameter uninitialized!";
156  }
157  }
158  }
159 }
