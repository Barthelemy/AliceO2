134 void HitAnalysis::ProcessChips(){
135  // Add test: All chips must be empty
136  Int_t nchipsNotEmpty(0);
137  std::vector<int> nonEmptyChips;
138  for (auto chipiter: fChips) {
139  if (chipiter.second->GetNumberOfPoints() > 0) {
140  nonEmptyChips.push_back(chipiter.second->GetChipIndex());
141  nchipsNotEmpty++;
142  }
143  }
144  if (nchipsNotEmpty) {
145  LOG(ERROR) << "Number of non-empty chips larger 0: " << nchipsNotEmpty << FairLogger::endl;
146  for (auto inditer : nonEmptyChips) {
147  LOG(ERROR) << "Chip index " << inditer << FairLogger::endl;
148  }
149  }
150 
151  // Assign hits to chips
152  for (TIter pointIter = TIter(fPointsArray).Begin(); pointIter != TIter::End(); ++pointIter) {
153  Point *point = static_cast<Point *>(*pointIter);
154  try{
155  fChips[point->GetDetectorID()]->InsertPoint(point);
156  } catch (Chip::IndexException &e) {
157  LOG(ERROR) << e.what() << FairLogger::endl;
158  }
159  }
160 
161  // Add test: Total number of hits assigned to chips must be the same as the size of the points array
162  Int_t nHitsAssigned(0);
163  for (auto chipiter : fChips) {
164  nHitsAssigned += chipiter.second->GetNumberOfPoints();
165  }
166  if (nHitsAssigned != fPointsArray->GetEntries()) {
167  LOG(ERROR) << "Number of points mismatch: Read(" << fPointsArray->GetEntries() << "), Assigned(" <<nHitsAssigned << ")" << FairLogger::endl;
168  }
169 
170  // Values for line segment calculation
171  double x0, x1, y0, y1, z0, z1, tof, edep, steplength;
172 
173  // loop over chips, get the line segment
174  for (auto chipiter: fChips) {
175  Chip &mychip = *(chipiter.second);
176  if (!mychip.GetNumberOfPoints()) {
177  continue;
178  }
179  //LOG(DEBUG) << "Processing chip with index " << mychip.GetChipIndex() << FairLogger::endl;
180  for (int ihit = 0; ihit < mychip.GetNumberOfPoints(); ihit++) {
181  if( mychip[ihit]->IsEntering() ) continue;
182  mychip.LineSegmentLocal(ihit, x0, x1, y0, y1, z0, z1, tof, edep);
183  steplength = TMath::Sqrt(x1*x1+y1*y1+z1*z1);
184  fLineSegment->Fill(steplength);
185  fLocalX0->Fill(x0);
186  fLocalX1->Fill(x1);
187  fLocalY0->Fill(y0);
188  fLocalY1->Fill(y1);
189  fLocalZ0->Fill(z0);
190  fLocalZ1->Fill(z1);
191  }
192  }
193  // Clear all chips
194  for (std::map<int, Chip*>::iterator chipiter = fChips.begin(); chipiter != fChips.end(); ++chipiter) {
195  chipiter->second->Clear();
196  }
197 }
