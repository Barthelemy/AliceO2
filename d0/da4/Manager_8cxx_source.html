144 {

146  // to the entries map and the ids to the ids list taking them from
147  // the map and the list found in the input file
148 
149  // if the manager is locked it cannot initialize from a snapshot
150  if (mLock) {
151  LOG(ERROR) << "Being locked I cannot initialize from the snapshot!" << FairLogger::endl;
152  return kFALSE;
153  }
154

156  TString snapshotFile(snapshotFileName);
157  if (snapshotFile.BeginsWith("alien://")) {
158  if (!gGrid) {
159  TGrid::Connect("alien://", "");

162  return kFALSE;

164  }
165  }

168  if (!f || f->IsZombie()) {
169  LOG(ERROR) << "Cannot open file " << snapshotFileName << FairLogger::endl;
170  return kFALSE;

172

174  TMap* entriesMap = 0;
175  TIter next(f->GetListOfKeys());
176  TKey* key;
177  while ((key = (TKey*)next())) {
178  if (strcmp(key->GetClassName(), "TMap") != 0)
179  continue;
180  entriesMap = (TMap*)key->ReadObj();
181  break;
182  }
183  if (!entriesMap || entriesMap->GetEntries() == 0) {
184  LOG(ERROR) << "Cannot get valid map of CDB entries from snapshot file" << FairLogger::endl;
185  return kFALSE;
186  }
187 
188  // retrieve ids' list from snapshot file
189  TList* idsList = 0;
190  TIter nextKey(f->GetListOfKeys());
191  TKey* keyN;
192  while ((keyN = (TKey*)nextKey())) {
193  if (strcmp(keyN->GetClassName(), "TList") != 0)
194  continue;
195  idsList = (TList*)keyN->ReadObj();
196  break;
197  }
198  if (!idsList || idsList->GetEntries() == 0) {
199  LOG(ERROR) << "Cannot get valid list of CDB entries from snapshot file" << FairLogger::endl;
200  return kFALSE;
201  }
202 
203  // Add each (entry,id) from the snapshot to the memory: entry to the cache, id to the list of ids.
204  // If "overwrite" is false: add the entry to the cache and its id to the list of ids
205  // only if neither of them is already there.
206  // If "overwrite" is true: write the snapshot entry,id in any case. If something
207  // was already there for that calibration type, remove it and issue a warning
208  TIter iterObj(entriesMap->GetTable());
209  TPair* pair = 0;
210  Int_t nAdded = 0;
211  while ((pair = dynamic_cast<TPair*>(iterObj.Next()))) {
212  TObjString* os = (TObjString*)pair->Key();
213  TString path = os->GetString();
214  TIter iterId(idsList);
215  ConditionId* id = 0;
216  ConditionId* correspondingId = 0;
217  while ((id = dynamic_cast<ConditionId*>(iterId.Next()))) {
218  TString idpath(id->getPathString());
219  if (idpath == path) {
220  correspondingId = id;
221  break;
222  }
223  }
224  if (!correspondingId) {
225  LOG(ERROR) << "id for \"" << path.Data()
226  << "\" not found in the snapshot (while entry was). This entry is skipped!" << FairLogger::endl;
227  break;
228  }
229  Bool_t cached = mConditionCache.Contains(path.Data());
230  Bool_t registeredId = kFALSE;
231  TIter iter(mIds);
232  ConditionId* idT = 0;
233  while ((idT = dynamic_cast<ConditionId*>(iter.Next()))) {
234  if (idT->getPathString() == path) {
235  registeredId = kTRUE;
236  break;
237  }
238  }
239 
240  if (overwrite) {
241  if (cached || registeredId) {
242  LOG(WARNING) << "An entry was already cached for \"" << path.Data()
243  << "\". Removing it before caching from snapshot" << FairLogger::endl;
244  unloadFromCache(path.Data());
245  }
246  mConditionCache.Add(pair->Key(), pair->Value());
247  mIds->Add(id);
248  nAdded++;
249  } else {
250  if (cached || registeredId) {
251  LOG(WARNING) << "An entry was already cached for \"" << path.Data()
252  << "\". Not adding this object from snapshot" << FairLogger::endl;
253  } else {
254  mConditionCache.Add(pair->Key(), pair->Value());
255  mIds->Add(id);
256  nAdded++;
257  }
258  }
259  }
260 
261  // mCondition is the new owner of the cache
262  mConditionCache.SetOwnerKeyValue(kTRUE, kTRUE);
263  entriesMap->SetOwnerKeyValue(kFALSE, kFALSE);
264  mIds->SetOwner(kTRUE);
265  idsList->SetOwner(kFALSE);
266  LOG(INFO) << nAdded << " new (entry,id) cached. Total number " << mConditionCache.GetEntries() << FairLogger::endl;

268  f->Close();
269  delete f;

670 void Manager::validateCvmfsCase() const
671 {
672  // The OCDB_PATH variable contains the path to the directory in /cvmfs/ which is
673  // an AliRoot tag based snapshot of the AliEn file catalogue (e.g.
674  // /cvmfs/alice.cern.ch/x86_64-2.6-gnu-4.1.2/Packages/OCDB/v5-05-76-AN).
675  // The directory has to contain:
676  // 1) <data|MC>/20??.list.gz gzipped text files listing the OCDB files (seen by that AliRoot tag)
677  // 2) bin/getOCDBFilesPerRun.sh (shell+awk) script extracting from 1) the list
678  // of valid files for the given run.
679 
680  if (!mCvmfsOcdb.BeginsWith("/cvmfs"))
681  LOG(FATAL) << "OCDB_PATH set to an invalid path: " << mCvmfsOcdb.Data() << FairLogger::endl;
682 
683  TString cvmfsUri(mCvmfsOcdb);
684  gSystem->ExpandPathName(cvmfsUri);
685  if (gSystem->AccessPathName(cvmfsUri))
686  LOG(FATAL) << "OCDB_PATH set to an invalid path: " << cvmfsUri.Data() << FairLogger::endl;
687 
688  // check that we find the two scripts we need
689 
690  LOG(DEBUG) << "OCDB_PATH envvar is set. Changing OCDB storage from alien:// to local:///cvmfs type."
691  << FairLogger::endl;
692  cvmfsUri = cvmfsUri.Strip(TString::kTrailing, '/');
693  cvmfsUri.Append("/bin/getOCDBFilesPerRun.sh");
