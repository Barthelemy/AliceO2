45 int Component::init(int argc, char** argv)
46 {
48 
49  // parse options
50  static struct option programOptions[] = {
51  {"library", required_argument, 0, 'l'},
52  {"component", required_argument, 0, 'c'},
53  {"parameter", required_argument, 0, 'p'},
54  {"run", required_argument, 0, 'r'},
55  {"msgsize", required_argument, 0, 's'},
56  {"output-mode", required_argument, 0, 'm'},
57  {"instance-id", required_argument, 0, 'i'},
58  {0, 0, 0, 0}
59  };
60 
61  /* getopt_long stores the option index here. */
62  char c = 0;
63  int iOption = 0;
64 
65  // HLT components are implemented in shared libraries, the library name
66  // and component id are used to factorize a component
67  // optionally, a list of configuration parameters can be specified as
68  // one single string which is translated in an array of string in the
69  // argc/argv format
70  const char* componentLibrary = "";
71  const char* componentId = "";
72  const char* componentParameter = "";
73  const char* instanceId="";
74 
75  // the configuration and calibration is fixed for every run and identified
76  // by the run no
77  int runNumber = 0;
78 
79  optind = 1; // indicate new start of scanning, especially when getop has been used in a higher layer already
80  while ((c = getopt_long(argc, argv, "l:c:p:r:s:m:i:", programOptions, &iOption)) != -1) {
81  switch (c) {
82  case 'l':
83  componentLibrary = optarg;
84  break;
85  case 'c':
86  componentId = optarg;
87  break;
88  case 'p':
89  componentParameter = optarg;
90  break;
91  case 'r':
92  std::stringstream(optarg) >> runNumber;
93  break;
94  case 's': {
95  unsigned size = 0;
96  std::stringstream(optarg) >> size;
97  mOutputBuffer.resize(size);
98  } break;
99  case 'm': {
100  unsigned outputMode;
101  std::stringstream(optarg) >> outputMode;
102  mFormatHandler.setOutputMode(outputMode);
103  } break;
104  case 'i': {
105  instanceId=optarg;
106  break;
107  }
108  case '?':
109  // TODO: more error handling
110  break;
111  default:
112  cerr << "unknown option: '" << c << "'" << endl;
113  }
114  }
115 
116  if (componentLibrary == NULL || componentLibrary[0] == 0 || componentId == NULL || componentId[0] == 0 ||
117  runNumber < 0) {
118  cerr << "missing argument" << endl;
119  return -EINVAL;
120  }
121 
122  int iResult = 0;
123  // TODO: make the SystemInterface a singleton
124  unique_ptr<ALICE::HLT::SystemInterface> iface(new SystemInterface);
125  if (iface.get() == NULL || ((iResult = iface->initSystem(runNumber))) < 0) {
126  // LOG(ERROR) << "failed to set up SystemInterface " << iface.get() << " (" << iResult << ")";
127  return -ENOSYS;
128  }
129 
130  // basic initialization succeeded, make the instances persistent
131  mpSystem = iface.release();
132 
133  // load the component library
134  if ((iResult = mpSystem->loadLibrary(componentLibrary)) != 0) return iResult > 0 ? -iResult : iResult;
135 
136  // chop the parameter string in order to provide parameters in the argc/argv format
137  vector<const char*> parameters;
138  unsigned parameterLength = strlen(componentParameter);
139  unique_ptr<char> parameterBuffer(new char[parameterLength + 1]);
140  if (parameterLength > 0 && parameterBuffer.get() != NULL) {
141  strcpy(parameterBuffer.get(), componentParameter);
142  char* iterator = parameterBuffer.get();
143  parameters.push_back(iterator);
144  for (; *iterator != 0; iterator++) {
145  if (*iterator != ' ') continue;
146  *iterator = 0; // separate strings
147  if (*(iterator + 1) != ' ' && *(iterator + 1) != 0) parameters.push_back(iterator + 1);
148  }
149  }
150 
151  // create component
152  string description;
153  description+=" chainid="; description+=instanceId;
154  if ((iResult=mpSystem->createComponent(componentId, NULL, parameters.size(), &parameters[0], &mProcessor, description.c_str()))<0) {
155  // the ALICE HLT external interface uses the following error definition
156  // 0 success
157  // >0 error number
158  return iResult > 0 ? -iResult : iResult;
159  }
160 
161  mEventCount = 0;
162 
163  return iResult;
164 }

166 int Component::process(vector<MessageFormat::BufferDesc_t>& dataArray,
167  cballoc_signal_t* cbAllocate)
168 {
169  if (!mpSystem) return -ENOSYS;
170  int iResult = 0;
171 
172  unsigned outputBufferSize = 0;
173 
174  AliHLTComponentEventData evtData;
175  memset(&evtData, 0, sizeof(evtData));
176  evtData.fStructSize = sizeof(evtData);
177  if (mEventCount >= 0) {
178  // very simple approach to provide an event ID
179  // TODO: adjust to the relevant format if available
180  evtData.fEventID = mEventCount++;
181  }
182 
183  AliHLTComponentTriggerData trigData;
184  memset(&trigData, 0, sizeof(trigData));
185  trigData.fStructSize = sizeof(trigData);
186 
187  AliHLTUInt32_t outputBlockCnt = 0;
188  AliHLTComponentBlockData* pOutputBlocks = NULL;
189  AliHLTComponentEventDoneData* pEventDoneData = NULL;
190 
191  // prepare input structure for the ALICE HLT component
192  mFormatHandler.clear();
193  mFormatHandler.addMessages(dataArray);
194  vector<AliHLTComponentBlockData>& inputBlocks = mFormatHandler.getBlockDescriptors();
195  unsigned nofInputBlocks = inputBlocks.size();
196  if (dataArray.size() > 0 && nofInputBlocks == 0 && mFormatHandler.getEvtDataList().size() == 0) {
197  cerr << "warning: none of " << dataArray.size() << " input buffer(s) recognized as valid input" << endl;
198  }
199  dataArray.clear();
200 
201  if (mFormatHandler.getEvtDataList().size()>0) {
202  // copy the oldest event header
203  memcpy(&evtData, &mFormatHandler.getEvtDataList().front(), sizeof(AliHLTComponentEventData));
204  }
205 
206  // determine the total input size, needed later on for the calculation of the output buffer size
207  int totalInputSize = 0;
208  for (vector<AliHLTComponentBlockData>::const_iterator ci = inputBlocks.begin(); ci != inputBlocks.end(); ci++) {
209  totalInputSize += ci->fSize;
210  }
211 
212  // add event type data block
213  AliHLTComponentBlockData eventTypeBlock;
214  memset(&eventTypeBlock, 0, sizeof(eventTypeBlock));
215  eventTypeBlock.fStructSize = sizeof(eventTypeBlock);
216  // Note: no payload!
217  eventTypeBlock.fDataType = AliHLTComponentDataTypeInitializer("EVENTTYP", "PRIV");
218  eventTypeBlock.fSpecification = gkAliEventTypeData;
219  inputBlocks.push_back(eventTypeBlock);
220 
221  // process
222  evtData.fBlockCnt = inputBlocks.size();
223  int nofTrials = 2;
224  do {
225  unsigned long constEventBase = 0;
226  unsigned long constBlockBase = 0;
227  double inputBlockMultiplier = 0.;
228  mpSystem->getOutputSize(mProcessor, &constEventBase, &constBlockBase, &inputBlockMultiplier);
229  outputBufferSize = constEventBase + nofInputBlocks * constBlockBase + totalInputSize * inputBlockMultiplier;
230  outputBufferSize+=sizeof(AliHLTComponentStatistics) + sizeof(AliHLTComponentTableEntry);
231  // take the full available buffer and increase if that
232  // is too little
233  mOutputBuffer.resize(mOutputBuffer.capacity());
234  if (mOutputBuffer.size() < outputBufferSize) {
235  mOutputBuffer.resize(outputBufferSize);
236  } else if (nofTrials < 2) {
237  // component did not update the output size
238  break;
239  }
240  outputBufferSize = mOutputBuffer.size();
241  outputBlockCnt = 0;
242  // TODO: check if that is working with the corresponding allocation method of the
243  // component environment
244  if (pOutputBlocks) delete[] pOutputBlocks;
245  pOutputBlocks = NULL;
246  if (pEventDoneData) delete pEventDoneData;
247  pEventDoneData = NULL;
248 
249  iResult = mpSystem->processEvent(mProcessor, &evtData, &inputBlocks[0], &trigData,
250  &mOutputBuffer[0], &outputBufferSize,
251  &outputBlockCnt, &pOutputBlocks,
252  &pEventDoneData);
253  if (outputBufferSize > 0) {
254  if (outputBufferSize > mOutputBuffer.size()) {
255  cerr << "fatal error: component writing beyond buffer capacity" << endl;
256  return -EFAULT;
257  } else if (outputBufferSize < mOutputBuffer.size()) {
258  mOutputBuffer.resize(outputBufferSize);
259  }
260  } else {
261  mOutputBuffer.clear();
262  }
263 
264  } while (iResult == ENOSPC && --nofTrials > 0);
265 
266  // prepare output
267  if (outputBlockCnt >= 0) {
268  AliHLTUInt8_t* pOutputBufferStart = &mOutputBuffer[0];
269  AliHLTUInt8_t* pOutputBufferEnd = pOutputBufferStart + mOutputBuffer.size();
270  // consistency check for data blocks
271  // 1) all specified data must be either inside the output buffer given
272  // to the component or in one of the input buffers
273  // 2) filter out special data blocks if they have been forwarded
274  //
275  // Some components set both pointer and offset in the block descriptor as data
276  // reference for blocks in the output buffer. Due to a misinterpretation in the
277  // early days of development that is only allowed if fPtr is set to the beginning
278  // of output buffer. The block is however fully described by offset relative to
279  // beginning of output buffer.
280  unsigned validBlocks = 0;
281  unsigned totalPayloadSize = 0;
282  AliHLTComponentBlockData* pOutputBlock = pOutputBlocks;
283  AliHLTComponentBlockData* pFiltered = pOutputBlocks;
284  for (unsigned blockIndex = 0; blockIndex < outputBlockCnt; blockIndex++, pOutputBlock++) {
285  // filter special data blocks
286  if (pOutputBlock->fDataType == eventTypeBlock.fDataType) continue;
287 
288  // block descriptors without any attached payload are propagated
289  bool bValid = pOutputBlock->fSize == 0;
290 
291  // calculate the data reference
292  AliHLTUInt8_t* pStart =
293  pOutputBlock->fPtr != NULL ? reinterpret_cast<AliHLTUInt8_t*>(pOutputBlock->fPtr) : &mOutputBuffer[0];
294  pStart += pOutputBlock->fOffset;
295  AliHLTUInt8_t* pEnd = pStart + pOutputBlock->fSize;
296  pOutputBlock->fPtr = pStart;
297  pOutputBlock->fOffset = 0;
298 
299  // first search in the output buffer
300  bValid = bValid || pStart >= pOutputBufferStart && pEnd <= pOutputBufferEnd;
301 
302  // possibly a forwarded data block, try the input buffers
303  if (!bValid) {
304  vector<AliHLTComponentBlockData>::const_iterator ci = inputBlocks.begin();
305  for (; ci != inputBlocks.end(); ci++) {
306  AliHLTUInt8_t* pInputBufferStart = reinterpret_cast<AliHLTUInt8_t*>(ci->fPtr);
307  AliHLTUInt8_t* pInputBufferEnd = pInputBufferStart + ci->fSize;
308  if ((bValid = (pStart >= pInputBufferStart && pEnd <= pInputBufferEnd))) {
309  break;
310  }
311  }
312  }
313 
314  if (bValid) {
315  totalPayloadSize += pOutputBlock->fSize;
316  validBlocks++;
317  memcpy(pFiltered, pOutputBlock, sizeof(AliHLTComponentBlockData));
318  pFiltered++;
319  } else {
320  cerr << "Inconsistent data reference in output block " << blockIndex << endl;
321  }
322  }
323  evtData.fBlockCnt=validBlocks;
324 
325  // create the messages
326  // TODO: for now there is an extra copy of the data, but it should be
327  // handled in place
328  vector<MessageFormat::BufferDesc_t> outputMessages =
329  mFormatHandler.createMessages(pOutputBlocks, validBlocks, totalPayloadSize, evtData, cbAllocate);
330  dataArray.insert(dataArray.end(), outputMessages.begin(), outputMessages.end());
331  }
332 
333  // cleanup
334  // NOTE: don't cleanup mOutputBuffer as the data is going to be used outside the class
335  // until released.
336  inputBlocks.clear();
337  outputBlockCnt = 0;
338  if (pOutputBlocks) delete[] pOutputBlocks;
339  pOutputBlocks = NULL;
340  if (pEventDoneData) delete pEventDoneData;
341  pEventDoneData = NULL;
342 
343  return -iResult;
344 }
