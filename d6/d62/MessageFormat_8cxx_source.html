59 int MessageFormat::addMessage(AliHLTUInt8_t* buffer, unsigned size)
60 {
61  // add message
62  // this will extract the block descriptors from the message
63  // the descriptors refer to data in the original message buffer
64 
65  unsigned count = mBlockDescriptors.size();
66  // the buffer might start with an event descriptor of type AliHLTComponentEventData
67  // the first read attempt is assuming the descriptor and checking consistency
68  // - buffer size at least size of AliHLTComponentEventData struct
69  // - fStructSize member matches
70  // - number of blocks matches fBlockCnt member
71  unsigned position=0;
72  AliHLTComponentEventData* evtData=reinterpret_cast<AliHLTComponentEventData*>(buffer+position);
73  if (position+sizeof(AliHLTComponentEventData)<=size &&
74  evtData->fStructSize==sizeof(AliHLTComponentEventData)) {
75  position += sizeof(AliHLTComponentEventData);
76  } else {
77  // one of the criteria does not match -> no event descriptor
78  evtData=NULL;
79  }
80  do {
81  if (evtData && evtData->fBlockCnt==0 && size<sizeof(AliHLTComponentBlockData)) {
82  // special case: no block data, only event header
83  break;
84  }
85  if (readBlockSequence(buffer+position, size-position, mBlockDescriptors) < 0 ||
86  (evtData!=NULL && ((mBlockDescriptors.size()-count) != evtData->fBlockCnt))) {
87  // not in the format of a single block, check if its a HOMER block
88  if (readHOMERFormat(buffer+position, size-position, mBlockDescriptors) < 0 ||
89  (evtData!=NULL && ((mBlockDescriptors.size()-count) != evtData->fBlockCnt))) {
90  // not in HOMER format either
91  if (position>0) {
92  // try once more without the assumption of event data header
93  position=0;
94  evtData=NULL;
95  continue;
96  }
97  return -ENODATA;
98  }
99  }
100  } while (0);
101 
102  int result=0;
103  if (evtData && (result=insertEvtData(*evtData))<0) {
104  // error in the event data header, probably headers of different events
105  mBlockDescriptors.resize(count);
106  return result;
107  }
108 
109  return mBlockDescriptors.size() - count;
110 }

113 {

116  int i = 0;
117  for (vector<BufferDesc_t>::const_iterator data = list.begin(); data != list.end(); data++, i++) {

133 int MessageFormat::readBlockSequence(AliHLTUInt8_t* buffer, unsigned size,
134  vector<AliHLTComponentBlockData>& descriptorList) const
135 {
136  // read a sequence of blocks consisting of AliHLTComponentBlockData followed by payload
137  // from a buffer
138  if (buffer == NULL) return 0;
139  unsigned position = 0;
140  vector<AliHLTComponentBlockData> input;
141  while (position + sizeof(AliHLTComponentBlockData) < size) {
142  AliHLTComponentBlockData* p = reinterpret_cast<AliHLTComponentBlockData*>(buffer + position);
143  if (p->fStructSize == 0 || // no valid header
144  p->fStructSize + position > size || // no space for the header
145  p->fStructSize + p->fSize + position > size) { // no space for the payload
146  // the buffer is only a valid sequence of data blocks if payload
147  // of the last block exacly matches the buffer boundary
148  // otherwize all blocks added until now are ignored
149  return -ENODATA;
150  }
151  // insert a new block
152  input.push_back(*p);
153  position += p->fStructSize;
154  if (p->fSize > 0) {
155  input.back().fPtr = buffer + position;
156  position += p->fSize;
157  } else {
158  // Note: also a valid block, payload is optional
159  input.back().fPtr = NULL;
160  }
161  // offset always 0 for iput blocks
162  input.back().fOffset = 0;
163  }
164 
165  descriptorList.insert(descriptorList.end(), input.begin(), input.end());
166  return input.size();
167 }

169 int MessageFormat::readHOMERFormat(AliHLTUInt8_t* buffer, unsigned size,
170  vector<AliHLTComponentBlockData>& descriptorList) const
171 {
172  // read message payload in HOMER format
173  if (mpFactory == NULL) const_cast<MessageFormat*>(this)->mpFactory = new ALICE::HLT::HOMERFactory;
174  if (buffer == NULL || mpFactory == NULL) return -EINVAL;
175  unique_ptr<AliHLTHOMERReader> reader(mpFactory->OpenReaderBuffer(buffer, size));
176  if (reader.get() == NULL) return -ENOMEM;
177 
178  unsigned nofBlocks = 0;
179  if (reader->ReadNextEvent() == 0) {
180  nofBlocks = reader->GetBlockCnt();
181  for (unsigned i = 0; i < nofBlocks; i++) {
182  AliHLTComponentBlockData block;
183  memset(&block, 0, sizeof(AliHLTComponentBlockData));
184  block.fStructSize = sizeof(AliHLTComponentBlockData);
185  block.fDataType.fStructSize = sizeof(AliHLTComponentDataType);
186  homer_uint64 id = byteSwap64(reader->GetBlockDataType(i));
187  homer_uint32 origin = byteSwap32(reader->GetBlockDataOrigin(i));
188  memcpy(&block.fDataType.fID, &id,
189  sizeof(id) > kAliHLTComponentDataTypefIDsize ? kAliHLTComponentDataTypefIDsize : sizeof(id));
190  memcpy(&block.fDataType.fOrigin, &origin,
191  sizeof(origin) > kAliHLTComponentDataTypefOriginSize ? kAliHLTComponentDataTypefOriginSize : sizeof(origin));
192  block.fSpecification = reader->GetBlockDataSpec(i);
193  block.fPtr = const_cast<void*>(reader->GetBlockData(i));
194  block.fSize = reader->GetBlockDataLength(i);
195  descriptorList.push_back(block);
196  }
197  }
198 
199  return nofBlocks;
200 }

202 vector<MessageFormat::BufferDesc_t> MessageFormat::createMessages(const AliHLTComponentBlockData* blocks,
203  unsigned count, unsigned totalPayloadSize,
204  const AliHLTComponentEventData& evtData,
205  boost::signals2::signal<unsigned char* (unsigned int)> *cbAllocate)
206 {
207  const AliHLTComponentBlockData* pOutputBlocks = blocks;
208  AliHLTUInt32_t outputBlockCnt = count;
209  mDataBuffer.clear();
210  mMessages.clear();
211  if (mOutputMode == kOutputModeHOMER) {
212  AliHLTHOMERWriter* pWriter = createHOMERFormat(pOutputBlocks, outputBlockCnt);
213  if (pWriter) {
214  AliHLTUInt32_t position = mDataBuffer.size();
215  AliHLTUInt32_t offset = 0;
216  AliHLTUInt32_t payloadSize = pWriter->GetTotalMemorySize();
217  auto msgSize=payloadSize + sizeof(evtData);
218  auto pTarget=&mDataBuffer[position];
219  if (cbAllocate==nullptr) {
220  // make the target in the internal buffer
221  mDataBuffer.resize(position + msgSize);
222  } else {
223  // use callback to create target
224  pTarget=*(*cbAllocate)(msgSize);
225  if (pTarget==nullptr) {
226  throw std::bad_alloc();
227  }
228  }
229  memcpy(pTarget + offset, &evtData, sizeof(evtData));
230  offset+=sizeof(evtData);
231  pWriter->Copy(pTarget + offset, 0, 0, 0, 0);
232  mpFactory->DeleteWriter(pWriter);
233  offset+=payloadSize;
234  mMessages.push_back(MessageFormat::BufferDesc_t(pTarget, offset));
235  }
236  } else if (mOutputMode == kOutputModeMultiPart || mOutputMode == kOutputModeSequence) {
237  // the output blocks are assempled in the internal buffer, for each
238  // block BlockData is added as header information, directly followed
239  // by the block payload
240  //
241  // kOutputModeMultiPart:
242  // multi part mode adds one buffer descriptor per output block
243  // the devices decides what to do with the multiple descriptors, one
244  // option is to send them in a multi-part message
245  //
246  // kOutputModeSequence:
247  // sequence mode concatenates the output blocks in the internal
248  // buffer. In contrast to multi part mode, only one buffer descriptor
249  // for the complete sequence is handed over to device
250  AliHLTUInt32_t position = mDataBuffer.size();
251  auto pTarget=&mDataBuffer[position];
252  AliHLTUInt32_t offset = 0;
253  unsigned bi = 0;
254  const auto* pOutputBlock = pOutputBlocks;
255  auto maxBufferSize = sizeof(evtData) + count * sizeof(AliHLTComponentBlockData) + totalPayloadSize;
256  do {
257  if (bi == 0 ||
258  mOutputMode == kOutputModeMultiPart) {
259  // request a new message buffer when entering for the first time in concatanate mode
260  // and for every block in multi part mode
261  // the actual size depends on mode and block index
262  // - event data structure is only written at beginning of first
263  // message, regardsless of mode
264  // - concatanate mode requests one big buffer for sequential sequence of
265  // all blocks
266  auto msgSize = (bi==0?sizeof(evtData):0); // first message has event data
267  if (count>0 && mOutputMode==kOutputModeMultiPart) {
268  msgSize+=sizeof(AliHLTComponentBlockData) + pOutputBlock->fSize;
269  } else {
270  msgSize+=count*sizeof(AliHLTComponentBlockData) + totalPayloadSize;
271  }
272  if (cbAllocate==nullptr) {
273  // make the target in the internal buffer, for simplicity data is copied
274  // to a new buffer, a memmove would be possible to make room for block
275  // descriptors
276  if (mDataBuffer.size() < position + msgSize) {
277  if (bi != 0) {
278  throw std::runtime_error("complete buffer allocation must be done in the first cycle");
279  }
280  // resize to the full size at once
281  mDataBuffer.resize(position + maxBufferSize);
282  }
283  pTarget=&mDataBuffer[position];
284  } else {
285  // use callback to create target
286  pTarget=*(*cbAllocate)(msgSize);
287  if (pTarget==nullptr) {
288  throw std::bad_alloc();
289  }
290  }
291  offset=0;
292  }
293 
294  if (bi==0) {
295  // event data only in the first message in order to avoid increase of required
296  // buffer size due to duplicated event header
297  memcpy(pTarget + offset, &evtData, sizeof(evtData));
298  if (mOutputMode == kOutputModeMultiPart && evtData.fBlockCnt>1) {
299  // in multipart mode, there is only one block per part
300  // consequently, the number of blocks indicated in the event data header
301  // does not reflect the number of blocks in this data sample. But it is
302  // set to 1 to make the single message consistent
303  auto* pEvtData = reinterpret_cast<AliHLTComponentEventData*>(pTarget + offset);
304  pEvtData->fBlockCnt=1;
305  }
306  offset+=sizeof(evtData);
307  }
308 
309  if (bi<count) {
310  // copy BlockData and payload
311  AliHLTUInt8_t* pData = reinterpret_cast<AliHLTUInt8_t*>(pOutputBlock->fPtr);
312  pData += pOutputBlock->fOffset;
313  auto* bdTarget = reinterpret_cast<AliHLTComponentBlockData*>(pTarget + offset);
314  memcpy(bdTarget, pOutputBlock, sizeof(AliHLTComponentBlockData));
315  bdTarget->fOffset = 0;
316  bdTarget->fPtr = NULL;
317  offset += sizeof(AliHLTComponentBlockData);
318  memcpy(pTarget + offset, pData, pOutputBlock->fSize);
319  offset += pOutputBlock->fSize;
320  if (mOutputMode == kOutputModeMultiPart) {
321  // send one descriptor per block back to device
322  mMessages.push_back(MessageFormat::BufferDesc_t(pTarget, offset));
323  position+=offset;
324  }
325  pOutputBlock++;
326  }
327  }
328  while (++bi<count);
329  if (mOutputMode == kOutputModeSequence || count==0) {
330  // send one single descriptor for all concatenated blocks
331  mMessages.push_back(MessageFormat::BufferDesc_t(pTarget, offset));
332  }
333  } else {
334  // invalid output mode
335  cerr << "error ALICE::HLT::Component: invalid output mode " << mOutputMode << endl;
336  }
337  return mMessages;
338 }

340 AliHLTHOMERWriter* MessageFormat::createHOMERFormat(const AliHLTComponentBlockData* pOutputBlocks,
341  AliHLTUInt32_t outputBlockCnt) const
342 {
343  // send data blocks in HOMER format in one message
344  int iResult = 0;
345  if (mpFactory == NULL) const_cast<MessageFormat*>(this)->mpFactory = new ALICE::HLT::HOMERFactory;
346  if (!mpFactory) return NULL;
347  unique_ptr<AliHLTHOMERWriter> writer(mpFactory->OpenWriter());
348  if (writer.get() == NULL) return NULL;
349 
350  homer_uint64 homerHeader[kCount_64b_Words];
351  HOMERBlockDescriptor homerDescriptor(homerHeader);
352 
353  const AliHLTComponentBlockData* pOutputBlock = pOutputBlocks;
354  for (unsigned blockIndex = 0; blockIndex < outputBlockCnt; blockIndex++, pOutputBlock++) {
355  if (pOutputBlock->fPtr == NULL && pOutputBlock->fSize > 0) {
356  cerr << "warning: ignoring block " << blockIndex << " because of missing data pointer" << endl;
357  }
358  memset(homerHeader, 0, sizeof(homer_uint64) * kCount_64b_Words);
359  homerDescriptor.Initialize();
360  homer_uint64 id = 0;
361  homer_uint64 origin = 0;
362  memcpy(&id, pOutputBlock->fDataType.fID, sizeof(homer_uint64));
363  memcpy(((AliHLTUInt8_t*)&origin) + sizeof(homer_uint32), pOutputBlock->fDataType.fOrigin, sizeof(homer_uint32));
364  homerDescriptor.SetType(byteSwap64(id));
365  homerDescriptor.SetSubType1(byteSwap64(origin));
366  homerDescriptor.SetSubType2(pOutputBlock->fSpecification);
367  homerDescriptor.SetBlockSize(pOutputBlock->fSize);
368  writer->AddBlock(homerHeader, pOutputBlock->fPtr);
369  }
370  return writer.release();
371 }

373 int MessageFormat::insertEvtData(const AliHLTComponentEventData& evtData)
374 {
375  // insert event header to list, sort by time, oldest first
376  if (mListEvtData.size()==0) {
377  mListEvtData.push_back(evtData);
378  } else {
379  vector<AliHLTComponentEventData>::iterator it=mListEvtData.begin();
380  for (; it!=mListEvtData.end(); it++) {
381  if ((it->fEventCreation_us*1e3 + it->fEventCreation_us/1e3)>
382  (evtData.fEventCreation_us*1e3 + evtData.fEventCreation_us/1e3)) {
383  // found a younger element
384  break;
385  }
386  }
387  // TODO: simple logic at the moment, header is not inserted
388  // if there is a mismatch, as the headers are inserted one by one, all
389  // headers in the list have the same ID
390  if (it != mListEvtData.end() &&
391  evtData.fEventID!=it->fEventID) {
392  cerr << "Error: mismatching event ID " << evtData.fEventID
393  << ", expected " << it->fEventID
394  << " for event with timestamp "
395  << evtData.fEventCreation_us*1e3 + evtData.fEventCreation_us/1e3 << " ms"
396  << endl;
397  return -1;
398  }
399  // insert before the younger element
400  mListEvtData.insert(it, evtData);
401  }
402  return 0;
403 }

405 AliHLTUInt64_t MessageFormat::byteSwap64(AliHLTUInt64_t src) const
406 {
407  // swap a 64 bit number
408  return ((src & 0xFFULL) << 56) |
409  ((src & 0xFF00ULL) << 40) |
410  ((src & 0xFF0000ULL) << 24) |
411  ((src & 0xFF000000ULL) << 8) |
412  ((src & 0xFF00000000ULL) >> 8) |
413  ((src & 0xFF0000000000ULL) >> 24) |
414  ((src & 0xFF000000000000ULL) >> 40) |
415  ((src & 0xFF00000000000000ULL) >> 56);
