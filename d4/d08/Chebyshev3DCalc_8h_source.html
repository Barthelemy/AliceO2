180 inline Float_t Chebyshev3DCalc::chebyshevEvaluation1D(Float_t x, const Float_t* array, int ncf)
181 {
182  if (ncf <= 0) {
183  return 0;
184  }
185 
186  Float_t b0, b1, b2, x2 = x + x;
187  b0 = array[--ncf];
188  b1 = b2 = 0;
189 
190  for (int i = ncf; i--;) {
191  b2 = b1;
192  b1 = b0;
193  b0 = array[i] + x2 * b1 - b2;
194  }
195  return b0 - x * b1;
196 }

200 inline Float_t Chebyshev3DCalc::Eval(const Float_t* par) const
201 {
202  if (!mNumberOfRows) {
203  return 0.;
204  }
205  int ncfRC;
206  for (int id0 = mNumberOfRows; id0--;) {
207  int nCLoc = mNumberOfColumnsAtRow[id0]; // number of significant coefs on this row
208  int col0 = mColumnAtRowBeginning[id0]; // beginning of local column in the 2D boundary matrix
209  for (int id1 = nCLoc; id1--;) {
210  int id = id1 + col0;
211  mTemporaryCoefficients2D[id1] = (ncfRC = mCoefficientBound2D0[id])
212  ? chebyshevEvaluation1D(par[2], mCoefficients + mCoefficientBound2D1[id], ncfRC)
213  : 0.0;
214  }
215  mTemporaryCoefficients1D[id0] = nCLoc > 0 ? chebyshevEvaluation1D(par[1], mTemporaryCoefficients2D, nCLoc) : 0.0;
216  }
217  return chebyshevEvaluation1D(par[0], mTemporaryCoefficients1D, mNumberOfRows);
218 }
