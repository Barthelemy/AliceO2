888  LOG(FATAL) << "Wrong stave thickness for layer " << j << "(" << mStaveThickness[j] << ")"
889  << FairLogger::endl;
890  }
891  if (mTurboLayer[j] && mStaveWidth[j] <= 0) {
892  LOG(FATAL) << "Wrong stave width for layer " << j << "(" << mStaveWidth[j] << ")"
893  << FairLogger::endl;
894  }
895  if (mDetectorThickness[j] < 0) {
896  LOG(FATAL) << "Wrong chip thickness for layer " << j << "(" << mDetectorThickness[j] << ")"
897  << FairLogger::endl;
898  }
899 
900  if (j > 0) {
901  if (mLayerRadii[j] <= mLayerRadii[j - 1]) {
902  LOG(FATAL) << "Layer " << j << " radius (" << mLayerRadii[j] << ") is smaller than layer "
903  << j - 1 << " radius (" << mLayerRadii[j - 1] << ")" << FairLogger::endl;
904  }
905  }
906

908  LOG(INFO) << "Stave thickness for layer " << j << " not set, using default"

910  }
911  if (mDetectorThickness[j] == 0) {
912  LOG(INFO) << "Chip thickness for layer " << j << " not set, using default"
913  << FairLogger::endl;
914  }
915  }

918  TGeoVolume** wrapVols = 0;
919

921  wrapVols = new TGeoVolume* [mNumberOfWrapperVolumes];

940  mUpgradeGeometry[j] = new UpgradeV1Layer(j, kFALSE);
941  }

944  mUpgradeGeometry[j]->setRadius(mLayerRadii[j]);
945  mUpgradeGeometry[j]->setZLength(mLayerZLength[j]);
