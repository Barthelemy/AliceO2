412 void Stack::SelectTracks()
413 {
414 
415  // Clear storage map
416  mStoreMap.clear();
417 
418  // Check particles in the fParticle array
419  for (Int_t i = 0; i < mNumberOfEntriesInParticles; i++) {
420 
421  TParticle* thisPart = GetParticle(i);
422  Bool_t store = kTRUE;
423 
424  // Get track parameters
425  Int_t iMother = thisPart->GetMother(0);
426  TLorentzVector p;
427  thisPart->Momentum(p);
428  Double_t energy = p.E();
429  Double_t mass = p.M();
430  // Double_t mass = thisPart->GetMass();
431  Double_t eKin = energy - mass;
432 
433  // Calculate number of points
434  Int_t nPoints = 0;
435  for (Int_t iDet = kAliIts; iDet < kSTOPHERE; iDet++) {
436  pair<Int_t, Int_t> a(i, iDet);
437  if (mPointsMap.find(a) != mPointsMap.end()) {
438  nPoints += mPointsMap[a];
439  }
440  }
441 
442  // Check for cuts (store primaries in any case)
443  if (iMother < 0) {
444  store = kTRUE;
445  } else {
446  if (!mStoreSecondaries) {
447  store = kFALSE;
448  }
449  if (nPoints < mMinPoints) {
450  store = kFALSE;
451  }
452  if (eKin < mEnergyCut) {
453  store = kFALSE;
454  }
455  }
456 
457  // Set storage flag
458  mStoreMap[i] = store;
459  }
460 
461  // If flag is set, flag recursively mothers of selected tracks
462  if (mStoreMothers) {
463  for (Int_t i = 0; i < mNumberOfEntriesInParticles; i++) {
464  if (mStoreMap[i]) {
465  Int_t iMother = GetParticle(i)->GetMother(0);
466  while (iMother >= 0) {
467  mStoreMap[iMother] = kTRUE;
468  iMother = GetParticle(iMother)->GetMother(0);
469  }
470  }
471  }
472  }
473 }
