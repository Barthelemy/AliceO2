106 void WrapperDevice::Run()
107 {
109  int iResult=0;
110 
111 #ifdef USE_CHRONO
112  static system_clock::time_point refTime = system_clock::now();
113 #endif // USE_CHRONO
114 
115  FairMQPoller* poller = fTransportFactory->CreatePoller(fChannels["data-in"]);
116 
117  // inherited variables of FairMQDevice:
118  // fChannels
119  // fTransportFactory
120  int numInputs = fChannels["data-in"].size();
121  int NoOfMsgParts=numInputs-1;
122  int errorCount=0;
123  const int maxError=10;
124 
125  vector</*const*/ FairMQMessage*> inputMessages;
126  vector<int> inputMessageCntPerSocket(numInputs, 0);
127  int nReadCycles=0;
128  while (CheckCurrentState(RUNNING)) {
129 
130  // read input messages
131  poller->Poll(mPollingPeriod);
132  int inputsReceived=0;
133  bool receivedAtLeastOneMessage=false;
134  for(int i = 0; i < numInputs; i++) {
135  if (inputMessageCntPerSocket[i]>0) {
136  inputsReceived++;
137  continue;
138  }
139  if (poller->CheckInput(i)) {
140  do {
141  unique_ptr<FairMQMessage> msg(fTransportFactory->CreateMessage());
142  if (fChannels.at("data-in").at(i).Receive(msg.get())) {
143  receivedAtLeastOneMessage = true;
144  inputMessages.push_back(msg.release());
145  if (inputMessageCntPerSocket[i] == 0)
146  inputsReceived++; // count only the first message on that socket
147  inputMessageCntPerSocket[i]++;
148  if (mVerbosity > 3) {
149  LOG(INFO) << " |---- receive Msg from socket " << i;
150  }
151  }
152  } while (fChannels.at("data-in").at(i).ExpectsAnotherPart());
153  if (mVerbosity > 2) {
154  LOG(INFO) << "------ received " << inputMessageCntPerSocket[i] << " message(s) from socket " << i;
155  }
156  }
157  }
158  if (receivedAtLeastOneMessage) nReadCycles++;
159  if (inputsReceived<numInputs) {
160  continue;
161  }
162  mNSamples++;
163  mTotalReadCycles+=nReadCycles;
164  if (mMaxReadCycles<0 || mMaxReadCycles<nReadCycles)
165  mMaxReadCycles=nReadCycles;
166  // if (nReadCycles>1) {
167  // LOG(INFO) << "------ recieved complete Msg from " << numInputs << " input(s) after " << nReadCycles << " read cycles" ;
168  // }
169  nReadCycles=0;
170 #ifdef USE_CHRONO
171  auto duration = std::chrono::duration_cast<TimeScale>(std::chrono::system_clock::now() - refTime);
172 
173  if (mLastSampleTime>=0) {
174  int sampleTimeDiff=duration.count()-mLastSampleTime;
175  if (mMinTimeBetweenSample < 0 || sampleTimeDiff<mMinTimeBetweenSample)
176  mMinTimeBetweenSample=sampleTimeDiff;
177  if (mMaxTimeBetweenSample < 0 || sampleTimeDiff>mMaxTimeBetweenSample)
178  mMaxTimeBetweenSample=sampleTimeDiff;
179  }
180  mLastSampleTime=duration.count();
181  if (duration.count()-mLastCalcTime>fLogIntervalInMs) {
182  LOG(INFO) << "------ processed " << mNSamples << " sample(s) - total "
183  << mComponent->getEventCount() << " sample(s)";
184  if (mNSamples > 0) {
185  LOG(INFO) << "------ min " << mMinTimeBetweenSample << "ms, max " << mMaxTimeBetweenSample << "ms avrg "
186  << (duration.count() - mLastCalcTime) / mNSamples << "ms ";
187  LOG(INFO) << "------ avrg number of read cycles " << mTotalReadCycles / mNSamples
188  << " max number of read cycles " << mMaxReadCycles;
189  }
190  mNSamples=0;
191  mTotalReadCycles=0;
192  mMinTimeBetweenSample=-1;
193  mMaxTimeBetweenSample=-1;
194  mMaxReadCycles=-1;
195  mLastCalcTime=duration.count();
196  }
197 #endif //USE_CHRONO
198 
199  if (!mSkipProcessing) {
200  // prepare input from messages
201  vector<AliceO2::AliceHLT::MessageFormat::BufferDesc_t> dataArray;
202  for (vector</*const*/ FairMQMessage*>::iterator msg=inputMessages.begin();
203  msg!=inputMessages.end(); msg++) {
204  void* buffer=(*msg)->GetData();
205  dataArray.push_back(AliceO2::AliceHLT::MessageFormat::BufferDesc_t(reinterpret_cast<unsigned char*>(buffer), (*msg)->GetSize()));
206  }
207 
208  // create a signal with the callback to the buffer allocation, the component
209  // can create messages via the callback and writes data directly to buffer
210  cballoc_signal_t cbsignal;
211  cbsignal.connect([this](unsigned int size){return this->createMessageBuffer(size);} );
212  mMessages.clear();
213 
214  // call the component
215  if ((iResult=mComponent->process(dataArray, &cbsignal))<0) {
216  LOG(ERROR) << "component processing failed with error code " << iResult;
217  }
218 
219  // build messages from output data
220  if (dataArray.size() > 0) {
221  if (mVerbosity > 2) {
222  LOG(INFO) << "processing " << dataArray.size() << " buffer(s)";
223  }
224  for (auto opayload : dataArray) {
225  FairMQMessage* omsg=nullptr;
226  // loop over pre-allocated messages
227  for (auto premsg = begin(mMessages); premsg != end(mMessages); premsg++) {
228  if ((*premsg)->GetData() == opayload.mP &&
229  (*premsg)->GetSize() == opayload.mSize) {
230  omsg=*premsg;
231  if (mVerbosity > 2) {
232  LOG(DEBUG) << "using pre-allocated message of size " << opayload.mSize;
233  }
234  break;
235  }
236  }
237  if (omsg==nullptr) {
238  unique_ptr<FairMQMessage> msg(fTransportFactory->CreateMessage());
239  if (msg.get()) {
240  msg->Rebuild(opayload.mSize);
241  if (msg->GetSize() < opayload.mSize) {
242  iResult = -ENOSPC;
243  break;
244  }
245  if (mVerbosity > 2) {
246  LOG(DEBUG) << "scheduling message of size " << opayload.mSize;
247  }
248  AliHLTUInt8_t* pTarget = reinterpret_cast<AliHLTUInt8_t*>(msg->GetData());
249  memcpy(pTarget, opayload.mP, opayload.mSize);
250  mMessages.push_back(msg.release());
251  } else {
252  if (errorCount == maxError && errorCount++ > 0)
253  LOG(ERROR) << "persistent error, suppressing further output";
254  else if (errorCount++ < maxError)
255  LOG(ERROR) << "can not get output message from framework";
256  iResult = -ENOMSG;
257  }
258  }
259  }
260  }
261 
262  if (mMessages.size()>0) {
263  if (fChannels.find("data-out") != fChannels.end() && fChannels["data-out"].size() > 0) {
264  for (auto sendmsg = begin(mMessages); sendmsg != end(mMessages); sendmsg++) {
265  if (sendmsg + 1 == end(mMessages)) {
266  // this is the last data block
267  if (mVerbosity > 2) {
268  LOG(DEBUG) << "sending last message, size " << (*sendmsg)->GetSize();
269  }
270  fChannels["data-out"].at(0).Send(*sendmsg);
271  } else {
272  if (mVerbosity > 2) {
273  LOG(DEBUG) << "sending multipart message, size " << (*sendmsg)->GetSize();
274  }
275  fChannels["data-out"].at(0).Send(*sendmsg, "snd-more");
276  }
277  }
278  } else {
279  if (errorCount == maxError && errorCount++ > 0)
280  LOG(ERROR) << "persistent error, suppressing further output";
281  else if (errorCount++ < maxError)
282  LOG(ERROR) << "no output slot available (" << (fChannels.find("data-out") == fChannels.end() ? "uninitialized" : "0 slots")
283  << ")";
284  }
285  for (auto sendmsg : mMessages) delete sendmsg;
286  mMessages.clear();
287  }
288  }
289 
290  // cleanup
291  for (vector<FairMQMessage*>::iterator mit=inputMessages.begin();
292  mit!=inputMessages.end(); mit++) {
293  delete *mit;
294  }
295  inputMessages.clear();
296  for (vector<int>::iterator mcit=inputMessageCntPerSocket.begin();
297  mcit!=inputMessageCntPerSocket.end(); mcit++) {
298  *mcit=0;
299  }
300  }
301 
302  delete poller;
303 }
