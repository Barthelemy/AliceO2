827 Int_t UpgradeGeometryTGeo::extractNumberOfChipsPerModule(Int_t lay, int& nrow) const
828 {
829  Int_t numberOfChips = 0;
830  char stavnam[30];
831  TGeoVolume* volLd = 0;
832 
833  if (!mModuleName.IsNull()) {
834  snprintf(stavnam, 30, "%s%d", getITSModulePattern(), lay);
835  volLd = gGeoManager->GetVolume(stavnam);
836  }
837  if (!volLd) { // no modules on this layer, check substaves
838  if (!mHalfStaveName.IsNull()) {
839  snprintf(stavnam, 30, "%s%d", getITSHalfStavePattern(), lay);
840  volLd = gGeoManager->GetVolume(stavnam);
841  }
842  }
843  if (!volLd) { // no substaves on this layer, check staves
844  snprintf(stavnam, 30, "%s%d", getITSStavePattern(), lay);
845  volLd = gGeoManager->GetVolume(stavnam);
846  }
847  if (!volLd) {
848  LOG(FATAL) << "can't find volume containing chips on layer " << lay << FairLogger::endl;
849  }
850 
851  // Loop on all stave nodes, count Chip volumes by checking names
852  Int_t nNodes = volLd->GetNodes()->GetEntries();
853 
854  double xmin = 1e9, xmax = -1e9, zmin = 1e9, zmax = -1e9;
855  double lab[3], loc[3] = { 0, 0, 0 };
856  double dx = -1, dz = -1;
857 
858  for (Int_t j = 0; j < nNodes; j++) {
859  // AliInfo(Form("L%d %d of %d %s %s ->
860  // %d",lay,j,nNodes,volLd->GetNodes()->At(j)->GetName(),GetITSChipPattern(),numberOfChips));
861  TGeoNodeMatrix* node = (TGeoNodeMatrix*)volLd->GetNodes()->At(j);
862  if (!strstr(node->GetName(), getITSChipPattern())) {
863  continue;
864  }
865  node->LocalToMaster(loc, lab);
866  if (lab[0] > xmax) {
867  xmax = lab[0];
868  }
869  if (lab[0] < xmin) {
870  xmin = lab[0];
871  }
872  if (lab[2] > zmax) {
873  zmax = lab[2];
874  }
875  if (lab[2] < zmin) {
876  zmin = lab[2];
877  }
878 
879  numberOfChips++;
880 
881  if (dx < 0) {
882  TGeoShape* chShape = node->GetVolume()->GetShape();
883  TGeoBBox* bbox = dynamic_cast<TGeoBBox*>(chShape);
884  if (!bbox) {
885  LOG(FATAL) << "Chip " << node->GetName() << " volume is of unprocessed shape "
886  << chShape->IsA()->GetName() << FairLogger::endl;
887  } else {
888  dx = 2 * bbox->GetDX();
889  dz = 2 * bbox->GetDZ();
890  }
891  }
892  }
893 
894  double spanX = xmax - xmin;
895  double spanZ = zmax - zmin;
896  nrow = TMath::Nint(spanX / dx + 1);
897  int ncol = TMath::Nint(spanZ / dz + 1);
898  if (nrow * ncol != numberOfChips) {
899  LOG(ERROR) << "Inconsistency between Nchips=" << numberOfChips << " and Nrow*Ncol=" << nrow
900  << "*" << ncol << "->" << nrow * ncol << FairLogger::endl
901  << "Extracted chip dimensions (x,z): " << dx << " " << dz
902  << " Module Span: " << spanX << " " << spanZ << FairLogger::endl;
903  }
904  return numberOfChips;
905 }
