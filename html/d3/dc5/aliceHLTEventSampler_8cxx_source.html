449 int preprocessSocketsDDS(vector<SocketProperties_t>& sockets, std::string networkPrefix)
450 {
451  // binding sockets
452  // - required arguments: size, type, property name, min, max port
453  // - find available port within given range
454  // - create address
455  // - send property to DDS
456  //
457  // connecting sockets
458  // - required arguments: size, property name, count
459  // - replicate according to count
460  // - fetch property from DDS
461  // - if local, only treat properties from tasks on the same node
462  // - translate type from type of opposite socket
463  // - add address
464  int iResult=0;
465  vector<SocketProperties_t> ddsduplicates;
466  for (vector<SocketProperties_t>::iterator sit=sockets.begin();
467  sit!=sockets.end(); sit++) {
468  if (sit->method.compare("bind")==0) {
469  unsigned maskRequiredParams=(0x1<<SIZE)|(0x1<<TYPE)|(0x1<<PROPERTY)|(0x1<<MINPORT);
470  if ((sit->validParams&maskRequiredParams)!=maskRequiredParams) {
471  cerr << buildSocketParameterErrorMsg(maskRequiredParams, sit->validParams, "Error: missing parameter(s) for binding socket") << endl;
472  iResult=-1;
473  break;
474  }
475  // the port will be selected by the FairMQ framework during the
476  // bind process, the address is a placeholder at the moment
477  sit->address="tcp://"+networkPrefix+":";
478  } else if (sit->method.compare("connect")==0) {
479  unsigned maskRequiredParams=(0x1<<SIZE)|(0x1<<PROPERTY)|(0x1<<COUNT);
480  if ((sit->validParams&maskRequiredParams)!=maskRequiredParams) {
481  cerr << buildSocketParameterErrorMsg(maskRequiredParams, sit->validParams, "Error: missing parameter(s) for connecting socket") << endl;
482  iResult=-1;
483  break;
484  }
485  // the port adress will be read from DDS properties
486  // address initialized to the host ip/name which is then used to match
487  // properties on the same hostwhen using local mode
488  // the actual number of input ports is spefified by the count argument, the
489  // corresponding number of properties is expected from DDS
490  sit->address=networkPrefix;
491  // add n-1 duplicates of the port configuration
492  for (int i=0; i<sit->ddscount-1; i++) {
493  ddsduplicates.push_back(*sit);
494  ddsduplicates.back().ddscount=0;
495  }
496  } else {
497  cerr << "Error: invalid socket method '" << sit->method << "'" << endl;
498  iResult=-1; // TODO: find error codes
499  break;
500  }
501  }
502 
503  if (iResult>=0) {
504  sockets.insert(sockets.end(), ddsduplicates.begin(), ddsduplicates.end());
505  }
506  return iResult;
507 }

534 int readSocketPropertiesDDS(vector<SocketProperties_t>& sockets)
535 {
536  // read dds properties for connecting sockets
537  for (vector<SocketProperties_t>::iterator sit=sockets.begin();
538  sit!=sockets.end(); sit++) {
539  if (sit->method.compare("connect")==1) continue;
540  if (sit->ddscount==0) continue; // the previously inserted duplicates
541 
542 #ifdef ENABLE_DDS
543  dds::key_value::CKeyValue ddsKeyValue;
544  dds::key_value::CKeyValue::valuesMap_t values;
545 
546  std::string hostaddress=sit->address;
547  vector<SocketProperties_t>::iterator workit=sit;
548  int socketPropertiesToRead=sit->ddscount;
549  std::map<std::string, bool> usedProperties;
550  {
551  std::mutex keyMutex;
552  std::condition_variable keyCondition;
553 
554  ddsKeyValue.subscribe([&keyCondition](const string& _key, const string& _value) {keyCondition.notify_all();});
555 
556  for (int cycle=1; socketPropertiesToRead>0; cycle++) {
557  if (cycle>0) {
558  std::unique_lock<std::mutex> lock(keyMutex);
559  keyCondition.wait_until(lock, std::chrono::system_clock::now() + chrono::milliseconds(1000));
560  }
561  ddsKeyValue.getValues(sit->ddsprop.c_str(), &values);
562  cout << "Info: DDS getValues received " << values.size() << " value(s) of property " << sit->ddsprop
563  << " " << sit->ddscount-socketPropertiesToRead << " of " << sit->ddscount << " sockets processed" << endl;
564  for (dds::key_value::CKeyValue::valuesMap_t::const_iterator vit = values.begin();
565  vit!=values.end(); vit++) {
566  if (usedProperties.find(vit->first)!=usedProperties.end()) continue; // already processed
567  cout << "Info: processing property " << vit->first << ", value " << vit->second << " on host " << hostaddress << endl;
568  usedProperties[vit->first]=true;
569  if (!(sit->validParams&(0x1<<DDSGLOBAL))) {
570  std::string property=vit->second;
571  if (property.find(hostaddress)==std::string::npos) {
572  cout << "Info: local parameter requested, skipping " << vit->second << " on host " << hostaddress << endl;
573  continue;
574  }
575  }
576  if (workit==sockets.end()) break;
577  workit->address=vit->second;
578  cout << "DDS getValue:" << sit->ddsprop << " " << sit->ddscount-socketPropertiesToRead << ": " << workit->address << endl;
579  socketPropertiesToRead--;
580  while ((++workit)!=sockets.end()) {
581  if (workit->method.compare("connect")==1) continue;
582  if (workit->ddsprop!=sit->ddsprop) continue;
