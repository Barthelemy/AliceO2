57  // define some materials

60 
61  // define some media
62  TGeoMedium *Carbon = new TGeoMedium("C", 3, matCarbon);
63  TGeoMedium *Vacuum = new TGeoMedium("Vacuum", 4, matVacuum);
64 
65 
66  Int_t nSects=2;
67  Double_t z[] = { -100, 300}; // in cm
68  Double_t r[] = { 2.5, 2.5}; // in cm
69  Double_t Thickness = 0.05; // thickness of beam pipe [cm]
70  TGeoPcon* shape = new TGeoPcon(0., 360., nSects);
71  for (Int_t iSect = 0; iSect < nSects; iSect++) {
72  shape->DefineSection(iSect, z[iSect], r[iSect], r[iSect]+Thickness);
73  }
74 
75  // ---> Volume
76  TGeoVolume* pipe = new TGeoVolume("Pipe", shape, Carbon);
77 
78  // --Now create the same but diameter less by Thikness and vacuum instead of Carbon
79  TGeoPcon* Vshape = new TGeoPcon(0., 360., nSects);
80  for (Int_t iSect = 0; iSect < nSects; iSect++) {
81  Vshape->DefineSection(iSect, z[iSect], r[iSect], r[iSect]);
82  }
83 
84  // ---> Volume
85  TGeoVolume* Vpipe = new TGeoVolume("Pipe", shape, Vacuum);
86 
87  top->AddNode(pipe, 1);
88  top->AddNode(Vpipe, 1);
