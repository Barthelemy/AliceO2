278 void MagneticField::MachineField(const Double_t* x, Double_t* b) const
279 {
280  // ---- This is the ZDC part
281  // Compansators for Alice Muon Arm Dipole
282  const Double_t kBComp1CZ = 1075., kBComp1hDZ = 260. / 2., kBComp1SqR = 4.0 * 4.0;
283  const Double_t kBComp2CZ = 2049., kBComp2hDZ = 153. / 2., kBComp2SqR = 4.5 * 4.5;
284 
285  const Double_t kTripQ1CZ = 2615., kTripQ1hDZ = 637. / 2., kTripQ1SqR = 3.5 * 3.5;
286  const Double_t kTripQ2CZ = 3480., kTripQ2hDZ = 550. / 2., kTripQ2SqR = 3.5 * 3.5;
287  const Double_t kTripQ3CZ = 4130., kTripQ3hDZ = 550. / 2., kTripQ3SqR = 3.5 * 3.5;
288  const Double_t kTripQ4CZ = 5015., kTripQ4hDZ = 637. / 2., kTripQ4SqR = 3.5 * 3.5;
289 
290  const Double_t kDip1CZ = 6310.8, kDip1hDZ = 945. / 2., kDip1SqRC = 4.5 * 4.5, kDip1SqRA = 3.375 * 3.375;
291  const Double_t kDip2CZ = 12640.3, kDip2hDZ = 945. / 2., kDip2SqRC = 4.5 * 4.5, kDip2SqRA = 3.75 * 3.75;
292  const Double_t kDip2DXC = 9.7, kDip2DXA = 9.4;
293 
294  double rad2 = x[0] * x[0] + x[1] * x[1];
295 
296  b[0] = b[1] = b[2] = 0;
297 
298  // SIDE C
299  if (x[2] < 0.) {
300  if (TMath::Abs(x[2] + kBComp2CZ) < kBComp2hDZ && rad2 < kBComp2SqR) {
301  b[0] = mCompensatorField2C * mMultipicativeFactorDipole;
302  } else if (TMath::Abs(x[2] + kTripQ1CZ) < kTripQ1hDZ && rad2 < kTripQ1SqR) {
303  b[0] = mQuadrupoleGradient * x[1];
304  b[1] = mQuadrupoleGradient * x[0];
305  } else if (TMath::Abs(x[2] + kTripQ2CZ) < kTripQ2hDZ && rad2 < kTripQ2SqR) {
306  b[0] = -mQuadrupoleGradient * x[1];
307  b[1] = -mQuadrupoleGradient * x[0];
308  } else if (TMath::Abs(x[2] + kTripQ3CZ) < kTripQ3hDZ && rad2 < kTripQ3SqR) {
309  b[0] = -mQuadrupoleGradient * x[1];
310  b[1] = -mQuadrupoleGradient * x[0];
311  } else if (TMath::Abs(x[2] + kTripQ4CZ) < kTripQ4hDZ && rad2 < kTripQ4SqR) {
312  b[0] = mQuadrupoleGradient * x[1];
313  b[1] = mQuadrupoleGradient * x[0];
314  } else if (TMath::Abs(x[2] + kDip1CZ) < kDip1hDZ && rad2 < kDip1SqRC) {
315  b[1] = mDipoleField;
316  } else if (TMath::Abs(x[2] + kDip2CZ) < kDip2hDZ && rad2 < kDip2SqRC) {
317  double dxabs = TMath::Abs(x[0]) - kDip2DXC;
318  if ((dxabs * dxabs + x[1] * x[1]) < kDip2SqRC) {
319  b[1] = -mDipoleField;
320  }
321  }
322  }
323 
324  // SIDE A
325  else {
326  if (TMath::Abs(x[2] - kBComp1CZ) < kBComp1hDZ && rad2 < kBComp1SqR) {
327  // Compensator magnet at z = 1075 m
328  b[0] = mCompensatorField1A * mMultipicativeFactorDipole;
329  }
330 
331  if (TMath::Abs(x[2] - kBComp2CZ) < kBComp2hDZ && rad2 < kBComp2SqR) {
332  b[0] = mCompensatorField2A * mMultipicativeFactorDipole;
333  } else if (TMath::Abs(x[2] - kTripQ1CZ) < kTripQ1hDZ && rad2 < kTripQ1SqR) {
334  b[0] = -mQuadrupoleGradient * x[1];
335  b[1] = -mQuadrupoleGradient * x[0];
336  } else if (TMath::Abs(x[2] - kTripQ2CZ) < kTripQ2hDZ && rad2 < kTripQ2SqR) {
337  b[0] = mQuadrupoleGradient * x[1];
338  b[1] = mQuadrupoleGradient * x[0];
339  } else if (TMath::Abs(x[2] - kTripQ3CZ) < kTripQ3hDZ && rad2 < kTripQ3SqR) {
340  b[0] = mQuadrupoleGradient * x[1];
341  b[1] = mQuadrupoleGradient * x[0];
342  } else if (TMath::Abs(x[2] - kTripQ4CZ) < kTripQ4hDZ && rad2 < kTripQ4SqR) {
343  b[0] = -mQuadrupoleGradient * x[1];
344  b[1] = -mQuadrupoleGradient * x[0];
345  } else if (TMath::Abs(x[2] - kDip1CZ) < kDip1hDZ && rad2 < kDip1SqRA) {
346  b[1] = -mDipoleField;
347  } else if (TMath::Abs(x[2] - kDip2CZ) < kDip2hDZ && rad2 < kDip2SqRA) {
348  double dxabs = TMath::Abs(x[0]) - kDip2DXA;
349  if ((dxabs * dxabs + x[1] * x[1]) < kDip2SqRA) {
350  b[1] = mDipoleField;
351  }
352  }
353  }

450 MagneticField* MagneticField::createFieldMap(Float_t l3Cur, Float_t diCur, Int_t convention, Bool_t uniform,
451  Float_t beamenergy, const Char_t* beamtype, const Char_t* path)
452 {
453  const Float_t l3NominalCurrent1 = 30000.; // (A)
454  const Float_t l3NominalCurrent2 = 12000.; // (A)
455  const Float_t diNominalCurrent = 6000.; // (A)
456 
457  const Float_t tolerance = 0.03; // relative current tolerance
458  const Float_t zero = 77.; // "zero" current (A)
459 
460  BMap_t map = k5kG;
461  double sclL3, sclDip;
462 
463  Float_t l3Pol = l3Cur > 0 ? 1 : -1;
464  Float_t diPol = diCur > 0 ? 1 : -1;
465 
466  l3Cur = TMath::Abs(l3Cur);
467  diCur = TMath::Abs(diCur);
468 
469  if (TMath::Abs((sclDip = diCur / diNominalCurrent) - 1.) > tolerance && !uniform) {
470  if (diCur <= zero) {
471  sclDip = 0.; // some small current.. -> Dipole OFF
472  } else {
473  FairLogger::GetLogger()->Fatal(MESSAGE_ORIGIN, "Wrong dipole current (%f A)!", diCur);
474  }
475  }
476 
477  if (uniform) {
478  // special treatment of special MC with uniform mag field (normalized to 0.5 T)
479  // no check for scaling/polarities are done
480  map = k5kGUniform;
481  sclL3 = l3Cur / l3NominalCurrent1;
482  } else {
483  if (TMath::Abs((sclL3 = l3Cur / l3NominalCurrent1) - 1.) < tolerance) {
484  map = k5kG;
485  } else if (TMath::Abs((sclL3 = l3Cur / l3NominalCurrent2) - 1.) < tolerance) {
486  map = k2kG;
487  } else if (l3Cur <= zero && diCur <= zero) {
488  sclL3 = 0;
489  sclDip = 0;
490  map = k5kGUniform;
491  } else {
492  FairLogger::GetLogger()->Fatal(MESSAGE_ORIGIN, "Wrong L3 current (%f A)!", l3Cur);
493  }
494  }
495 
496  if (sclDip != 0 && map != k5kGUniform) {
497  if ((l3Cur <= zero) ||
498  ((convention == kConvLHC && l3Pol != diPol) || (convention == kConvDCS2008 && l3Pol == diPol))) {
499  FairLogger::GetLogger()->Fatal(MESSAGE_ORIGIN,
500  "Wrong combination for L3/Dipole polarities (%c/%c) for convention %d",
501  l3Pol > 0 ? '+' : '-', diPol > 0 ? '+' : '-', getPolarityConvention());
502  }
503  }
504 
505  if (l3Pol < 0) {
506  sclL3 = -sclL3;
507  }
508  if (diPol < 0) {
509  sclDip = -sclDip;
510  }
511 
512  BeamType_t btype = kNoBeamField;
513  TString btypestr = beamtype;
514  btypestr.ToLower();
515  TPRegexp protonBeam("(proton|p)\\s*-?\\s*\\1");
516  TPRegexp ionBeam("(lead|pb|ion|a|A)\\s*-?\\s*\\1");
517  TPRegexp protonionBeam("(proton|p)\\s*-?\\s*(lead|pb|ion|a|A)");
518  TPRegexp ionprotonBeam("(lead|pb|ion|a|A)\\s*-?\\s*(proton|p)");
519  if (btypestr.Contains(ionBeam)) {
520  btype = kBeamTypeAA;
521  } else if (btypestr.Contains(protonBeam)) {
522  btype = kBeamTypepp;
523  } else if (btypestr.Contains(protonionBeam)) {
524  btype = kBeamTypepA;
525  } else if (btypestr.Contains(ionprotonBeam)) {
526  btype = kBeamTypeAp;
527  } else {
528  FairLogger::GetLogger()->Info(MESSAGE_ORIGIN, "Assume no LHC magnet field for the beam type %s,", beamtype);
529  }
530  char ttl[80];
531  snprintf(ttl, 79, "L3: %+5d Dip: %+4d kA; %s | Polarities in %s convention", (int)TMath::Sign(l3Cur, float(sclL3)),
532  (int)TMath::Sign(diCur, float(sclDip)), uniform ? " Constant" : "",
533  convention == kConvLHC ? "LHC" : "DCS2008");
534  // LHC and DCS08 conventions have opposite dipole polarities
535  if (getPolarityConvention() != convention) {
536  sclDip = -sclDip;
537  }
538 
539  return new MagneticField("MagneticFieldMap", ttl, sclL3, sclDip, map, btype, beamenergy, 2, 10., path);
540 }
