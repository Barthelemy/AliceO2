157 void Chip::MedianHitGlobal(const Point *p1, const Point *p2, Double_t &x, Double_t &y, Double_t &z) const {
158  // Get hit positions in global coordinates
159  Double_t pos1Glob[3] = {p1->GetX(), p1->GetY(), p1->GetZ()},
160  pos2Glob[3] = {p2->GetX(), p2->GetY(), p2->GetZ()}, posMedianLocal[3], posMedianGlobal[3];
161 
162  // Calculate mean positions
163  posMedianLocal[1] = 0.;
164  if ((pos1Glob[1] * pos2Glob[1]) < 0.) {
165  posMedianLocal[0] = (-pos1Glob[1]/(pos2Glob[1] - pos1Glob[1])) * (pos2Glob[0] - pos1Glob[0]) + pos1Glob[0];
166  posMedianLocal[2] = (-pos1Glob[1]/(pos2Glob[1] - pos1Glob[1])) * (pos2Glob[2] - pos1Glob[2]) + pos1Glob[2];
167  } else {
168  posMedianLocal[0] = 0.5 * (pos1Glob[0] + pos2Glob[0]);
169  posMedianLocal[2] = 0.5 * (pos1Glob[2] + pos2Glob[2]);
170  }
171 
172  // Convert to global coordinates
173  fGeometry->localToGlobal(fChipIndex, posMedianLocal, posMedianGlobal);
174  x = posMedianGlobal[0];
